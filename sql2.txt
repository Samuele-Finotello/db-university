Query con JOIN:

1 - SELECT students.`id`, students.`name` AS `name_students`, students.`surname` AS `surname_students`, 
    students.`registration_number`, students.`enrolment_date`, degrees.`name` AS `name_degrees`
    FROM db_university.students
    INNER JOIN degrees ON students.`degree_id` = `degrees.id`
    WHERE degrees.`name` = "Corso di Laurea in Economia";

2 - SELECT departments.`name` AS `name_departments`, degrees.`name` AS `name_degrees`, degrees.`level`, departments.`address`,
    departments.`head_of_department`
    FROM db_university.departments
    JOIN degrees ON degrees.`department_id` = departments.`id`
    WHERE departments.`name` = "Dipartimento di Neuroscienze"
    AND degrees.`level` = "magistrale";
